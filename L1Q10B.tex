\begin{questions}
  
\question We are considering a proof by induction of the following claim.
  
  \centerline{$2^n < n!$ for every integer $n$ with $n\geq 4$.}

  For this particular proof,
  \begin{parts}
  \part[2] Write the statement as a proposition, $P(n)$, and indicate its domain.
    \begin{solution}
    Let $\mathbb{Z}'=\mathbb{Z}^+-\{1,2,3\}$. Then the statement is,
      \[
        P(n): 2^n < n! \; \text{ where } n\in\mathbb{Z}' .
      \]
    \end{solution}
    
  \part[2] What would need to be shown in the basis step?
    \begin{solution}
      It would need to be shown that $P(4)$ is true, i.e., that $2^4<4!$.
    \end{solution}
    
  \part[2] What does the inductive hypothesis claim?
    \begin{solution}
      It claims that $P(k)$ is true, i.e., $2^k < k! \; \text{ where } k\in\mathbb{Z}' $.
    \end{solution}
  \part[4] What does the inductive step need to show?
    \begin{solution}
      It needs to show that if $P(k)$ is true then $P(k+1)$ is also true, i.e.,
      \[
        (2^k < k!) \implies (2^{k+1} < (k+1)!) \; \text{ where } k\in\mathbb{Z}^*.
      \]
    \end{solution}
  \end{parts}
  Note that you are not being asked to execute the proof.

  
\end{questions}
