\begin{questions}
\question[10] Given
  \[
    f(n) = \sum_{i=1}^n k^2,
  \]
  show how you can compute $\sum_{i=10}^{100} k^2$.
 
  \begin{solution}
    \begin{align*}
      \sum_{i=10}^{100} k^2 & = \sum_{i=1}^{100} k^2 - \sum_{i=1}^{9} k^2\\
                            & = f(100) - f(9)
    \end{align*}
  \end{solution}
