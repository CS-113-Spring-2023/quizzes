\begin{questions}
  
\question We are considering a proof by induction of the following claim.
  
  \centerline{$7^{n+2} + 8^{2n+1}$ is divisible by $57$ for every non-negative integer $n$.}

  For this particular proof,
  \begin{parts}
  \part[2] Write the statement as a proposition, $P(n)$, and indicate its domain.
    \begin{solution}
      \[
        P(n): \exists m\in\mathbb{Z} \ni (7^{n+2} + 8^{2n+1} = 57m) \; \text{ where } n\in\mathbb{Z}^* .
      \]
    \end{solution}
    
  \part[2] What would need to be shown in the basis step?
    \begin{solution}
      It would need to be shown that $P(0)$ is true, i.e. that $7^2+8^1$ is divisible by 57.
    \end{solution}
    
  \part[2] What does the inductive hypothesis claim?
    \begin{solution}
      It claims that $P(k)$ is true, i.e., $\exists m\in\mathbb{Z} \ni (7^{k+2} + 8^{2k+1} = 57m) \; \text{ where } k\in\mathbb{Z}^* $.
    \end{solution}
  \part[4] What does the inductive step need to show?
    \begin{solution}
      It needs to show that if $P(k)$ is true then $P(k+1)$ is also true, i.e.,
      \[
        \exists m\in\mathbb{Z} \ni (7^{k+2} + 8^{2k+1} = 57m) \implies
        \exists m\in\mathbb{Z} \ni (7^{k+3} + 8^{2k+3} = 57m)
         \; \text{ where } k\in\mathbb{Z}^*.
      \]
    \end{solution}
  \end{parts}
  Note that you are not being asked to execute the proof.

  
\end{questions}
